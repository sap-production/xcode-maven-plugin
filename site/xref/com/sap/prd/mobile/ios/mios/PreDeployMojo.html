44          "In case this does not work click <a href=\"$LOCATION\">here</a></body></html>";

46    @Override
47    public void execute() throws MojoExecutionException, MojoFailureException
48    {
49      try {
50        archiveFolder = archiveFolder.getCanonicalFile();
51      }
52      catch (IOException ex) {
53        throw new MojoExecutionException(ex.getMessage(), ex);
54      }
55  
56      createArchiveFolder(archiveFolder);
57  
58      getLog().info("Using archive directory: '" + archiveFolder + "'.");
59  
60      try {
61  
62        //
63        // [Q] Why do we use reflection here? What about a simple downcast to DefaultRepositorySystemSession?
64        // [A] We cannot downcast since the downcast fails. The DefaultRepositorySystemSession that is available
65        //     here is loaded by another class loader than the DefaultRepositorySystemSession used by the actual class.
66        //     Hence we are in different runtime packages and the downcast is not possible.
67  
68        final TransferListener transferListener = (TransferListener) this.repoSession.getClass()
69          .getMethod("getTransferListener", new Class[0]).invoke(this.repoSession, new Object[0]);
70  
71        TransferListener prepareIpaPointerFileTransferListener = new PrepareIpaPointerFileTransferListener(
72              transferListener);
73  
74        this.repoSession.getClass().getMethod("setTransferListener", new Class[] { TransferListener.class })
75          .invoke(this.repoSession, prepareIpaPointerFileTransferListener);
76  
77        getLog().info(
78              "TransferListener '" + prepareIpaPointerFileTransferListener.getClass().getName() + "' has been set.");
79      }
80      catch (RuntimeException e) {
81        throw e;
82      }
83      catch (Exception e) {
84        throw new MojoExecutionException(
85              "Cannot set transfer listener for creating artifact redirect HTML files: " + e.getMessage(), e);
86      }
87    }

89    private static void createArchiveFolder(File archiveFolder) throws MojoExecutionException
90    {
91      if (archiveFolder.isFile())

107     @Override
108     public void transferSucceeded(TransferEvent event)
109     {
110       if (forward != null)
111         forward.transferSucceeded(event);
112 
113       
114       if(event.getResource().getResourceName().endsWith("maven-metadata.xml")) {
115         
116         getLog().debug("No redirect html file will be created for '" + event.getResource().getResourceName() + "'.");
117         return;
118       }
119       
120       try {
121 
122         final String url = event.getResource().getRepositoryUrl() + event.getResource().getResourceName();
123         final String html = HTML_TEMPLATE.replaceAll("\\$LOCATION", url);
124 
125         final File redirect = new File(new File(new File(new File(archiveFolder, "artifacts"), project.getGroupId()),
126               project.getArtifactId()), getArtifactRedirectHtmlFileName(url));

128         writeArtifactRedirectHtmlFile(redirect, url, html);

131 
132       }
133       catch (RuntimeException ex) {

136           .error("Could not create artifact redirect HTML file for '" + event.getResource().getResourceName(), ex);
137       }

175     private void writeArtifactRedirectHtmlFile(final File redirect, final String url, final String html)
176     {
177       FileOutputStream fos = null;
178 
179       try {
180 
181         if (!redirect.getParentFile().exists() && !redirect.getParentFile().mkdirs())
182           throw new IOException("Cannot create folder '" + redirect.getParentFile() + "'.");
183 
184         fos = new FileOutputStream(redirect);
185 
186         IOUtils.write(html, fos, "UTF-8");
187       }
188       catch (IOException ex) {
189         throw new RuntimeException(ex);
190       }
191       finally {
192         IOUtils.closeQuietly(fos);
193       }
194     }

196     private String getArtifactRedirectHtmlFileName(final String url)
197     {
198       //E.g. MyApp-1.0.0-20120821.132955-1-Release-iphoneos-ota.htm 
199       String artifactFileName = url.substring(url.lastIndexOf("/") + 1);
200       return getRedirectHtmlFilename(artifactFileName, project.getArtifactId());

205   static String getRedirectHtmlFilename(String artifactFileName, String artifactId)
206   {
207     String search = "-";
208     int nthElement = 0;
209     if (artifactFileName.endsWith(XCodeOtaHtmlGeneratorMojo.OTA_CLASSIFIER_APPENDIX + "."
210           + XCodeOtaHtmlGeneratorMojo.OTA_HTML_FILE_APPENDIX)) {
211       nthElement = 3;
212     }
213     else if (artifactFileName.endsWith("-AppStoreMetaData.zip")) {
214       nthElement = 1;
215     }
216     else if (artifactFileName.endsWith("-app.dSYM.zip")) {
217       nthElement = 3;
218     }
219     else if (artifactFileName.endsWith("-app.zip")) {
220       nthElement = 3;
221     }
222     else if (artifactFileName.endsWith(".ipa")) {
223       nthElement = 2;
224     }
225     else if (artifactFileName.endsWith("versions.xml")) {
226       nthElement = 1;
227     }
228     else if (artifactFileName.endsWith(".pom")) {
229       nthElement = 1;
230       search = ".";
231     }
232 
233     int idx = getNthIndexFromBack(artifactFileName, search, nthElement);
234     if (idx >= 0) {
235       String name = artifactId + artifactFileName.substring(idx);
236       if (!name.endsWith(".htm")) {
237         name = name + ".htm";
238       }
239       return name;
240     }
241     return artifactId + "-" + artifactFileName + ".htm";
242   }
243 
244   static int getNthIndexFromBack(String string, String searchString, int countFromBack)
245   {
246     if (countFromBack <= 0) return -1;
247     int idx = string.length();
248     for (int i = 0; i < countFromBack; i++) {
249       idx = string.substring(0, idx).lastIndexOf(searchString);
250       if (idx <= 0) return idx;
251     }
252     return idx;
253   }
