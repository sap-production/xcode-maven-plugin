20  package com.sap.prd.mobile.ios.mios;
21  
22  import java.io.File;
23  
24  import org.apache.maven.artifact.Artifact;
25  import org.apache.maven.project.MavenProject;
26  
27  /**
28   * Helper methods for Xcode build to retrieve certain directories below the target folder.
29   * 
30   */
31  class MavenTargetFolderLayout
32  {
33    private static String XCODE_DEPS_TARGET_FOLDER = "xcode-deps";
34  
35    private static File getFolderForExtractedHeaders(MavenProject project, final String configuration, final String sdk)
36    {
37      return new File(new File(new File(project.getBuild().getDirectory()), "headers"), configuration + "-" + sdk);
38    }
39  
40    private static File getFolderForExtractedLibs(MavenProject project, final String configuration, final String sdk)
41    {
42      return new File(new File(new File(project.getBuild().getDirectory()), "libs"), configuration + "-" + sdk);
43    }
44  
45    private static File getFolderForExtractedBundles(MavenProject project)
46    {
47      return new File(new File(project.getBuild().getDirectory()), "bundles");
48    }
49  
50    private static File getFolderForExtractedFrameworks(MavenProject project)
51    {
52      return new File(new File(new File(project.getBuild().getDirectory()), XCODE_DEPS_TARGET_FOLDER), "frameworks");
53    }
54  
55    static File getFolderForExtractedHeadersWithGA(MavenProject project, final String configuration, final String sdk,
56          final String groupId, final String artifactId)
57    {
58      return new File(new File(getFolderForExtractedHeaders(project, configuration, sdk), groupId), artifactId);
59    }
60  
61    static File getFolderForExtractedLibsWithGA(MavenProject project, final String configuration, final String sdk,
62          final String groupId, final String artifactId)
63    {
64      return new File(new File(getFolderForExtractedLibs(project, configuration, sdk), groupId), artifactId);
65    }
66  
67    static File getFolderForExtractedBundlesWithGA(MavenProject project, final String groupId, final String artifactId)
68    {
69      return new File(new File(getFolderForExtractedBundles(project), groupId), artifactId);
70    }
71  
72    static File getFolderForExtractedFrameworkswithGA(MavenProject project, final String groupId, final String artifactId)
73    {
74      return new File(new File(getFolderForExtractedFrameworks(project), groupId), artifactId);
75    }
76  
77    static File getFolderForExtractedMainArtifact(MavenProject project)
78    {
79      return new File(new File(project.getBuild().getDirectory()), "main.artifact");
80    }
81  
82    static File getFolderForExtractedPrimaryArtifact(MavenProject project, Artifact primaryArtifact)
83    {
84      return new File(new File(new File(new File(project.getBuild().getDirectory()), "extractedPrimaryArtifacts"),
85            primaryArtifact.getGroupId()), primaryArtifact.getArtifactId());
86    }
87  }
