14  /**
15   * Assures that the following required prerequisites are fulfilled:
16   * <ul>
17   * <li>Xcode version {@value #MIN_XCODE_VERSION} or higher is installed</li>
18   * </ul>
19   * 
20   * @goal check-prerequisites
21   * 
22   */
23  public class CheckPrerequisitesMojo extends AbstractXCodeMojo
24  {
25    
26    public final static String MIN_XCODE_VERSION = "4.4";
27    
28    @Override
29    public void execute() throws MojoExecutionException, MojoFailureException
30    {
31      try {
32        ByteArrayOutputStream bos = new ByteArrayOutputStream();
33        PrintStream out = new PrintStream(bos);
34        int exitCode = Forker.forkProcess(out, new File("."), new String[] { "xcodebuild", "-version" });
35        if (exitCode == 0) {
36          String output = bos.toString();
37          DefaultArtifactVersion version = getVersion(output);
38          String buildVersion = getBuildVersion(output);
39          getLog().info("Using Xcode " + version + " " + buildVersion);
40          checkVersions(version, buildVersion);
41        }
42        else {
43          throw new MojoExecutionException("Could not get xcodebuild version (exit code = " + exitCode + ")");
44        }
45      }
46      catch (IOException e) {
47        throw new MojoExecutionException("Could not get xcodebuild version", e);
48      }
49    }
50  
51    private DefaultArtifactVersion getVersion(String output) throws MojoExecutionException
52    {
53      Pattern versionPattern = Pattern.compile("Xcode (\\d+(\\.\\d+)+)", Pattern.CASE_INSENSITIVE);
54      Matcher versionMatcher = versionPattern.matcher(output);
55      if (versionMatcher.find()) {
56        return new DefaultArtifactVersion(versionMatcher.group(1));
57      }
58      throw new MojoExecutionException("Could not get xcodebuild version");
59    }
60    
61    private String getBuildVersion(String output) throws MojoExecutionException {
62      Pattern buildPattern = Pattern.compile("Build version (\\w+)", Pattern.CASE_INSENSITIVE);
63      Matcher buildMatcher = buildPattern.matcher(output);
64      if (buildMatcher.find()) {
65        return buildMatcher.group(1);
66      }
67      throw new MojoExecutionException("Could not get xcodebuild build version");
68    }
69    
70    private void checkVersions(DefaultArtifactVersion version, String buildVersion) throws MojoExecutionException
71    {
72      DefaultArtifactVersion minXcodeVersion = new DefaultArtifactVersion(MIN_XCODE_VERSION);
73      if (version.compareTo(minXcodeVersion) < 0) {
74        throw new MojoExecutionException("Xcode " + MIN_XCODE_VERSION + " (or higher) is required (installed: " + version + " " + buildVersion + ")");
75      }
76    }
