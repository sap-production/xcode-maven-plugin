26  // Format:  <groupId>:<artifactId>[:<extension>[:<classifier>]]:<version>
27  class GAVUtil
28  {
29    static String toColonNotation(String groupId, String artifactId, String version, String extension, String classifier)
30    {
31      // For our use case we expect all parameter to be not null (or empty).
32      // Our use case is handling side artifacts with requires a classifier an an extension.
33      ensureNotNullOrEmpty(groupId, "groupId");
34      ensureNotNullOrEmpty(artifactId, "artifactId");
35      ensureNotNullOrEmpty(version, "version");
36      ensureNotNullOrEmpty(extension, "extension");
37      ensureNotNullOrEmpty(classifier, "classifier");
38  
39      return groupId + ":" + artifactId + ":" + extension + ":" + classifier + ":" + version;
40    }
41  
42    private static void ensureNotNullOrEmpty(String str, String parameterName)
43    {
44      if (str == null || str.isEmpty())
45        throw new IllegalStateException("Value of parameter '" + parameterName + "' was null or empty.");
46  
47    }
48  
49    static Artifact getArtifact(String coords)
50    {
51      DefaultArtifact aetherArtifact = new DefaultArtifact(coords);
52      return new org.apache.maven.artifact.DefaultArtifact(aetherArtifact.getGroupId(), aetherArtifact.getArtifactId(),
53            aetherArtifact.getVersion(), null, aetherArtifact.getExtension(), aetherArtifact.getClassifier(), null) {
54  
55        // The DefaultArtifact created here is only used internally as a
56        // vehicle. Since we do not set a scope and an artifact handler
57        // these methods cannot work in a reasonable way. Instead returning
58        // null which might lead to problems later on we throw an
59        // exception right when the corresponding methods are called.
60  
61        @Override
62        public String getScope()
63        {
64          throw new UnsupportedOperationException();
65        }
66  
67        @Override
68        public ArtifactHandler getArtifactHandler()
69        {
70          throw new UnsupportedOperationException();
71        }
72      };
73    }
