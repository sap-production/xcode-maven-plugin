39    static File getBinary(final File buildDir, final String configuration, final String sdk, final String projectName)
40    {
41  
42      return new File(buildDir, configuration + "-" + sdk + "/lib" + projectName + ".a");
43    }
44  
45    static File getBundleDirectory(final File srcDir, final String bundleName)
46    {
47  
48      return new File(srcDir, bundleName + ".bundle");
49    }
50  
51    static File getHeadersDirectory(final File buildDir, final String configuration, final String sdk)
52    {
53  
54      return new File(new File(buildDir, configuration + "-" + sdk), "usr/local/include");
55  
56    }
57  
58    // TODO invent better method name
59    static File getAppFolder(final File baseDirectory, final String configuration, final String sdk)
60    {
61      return new File(getBuildDir(baseDirectory), configuration + "-" + sdk);
62    }
63  
64    static File getBuildDir(final File baseDirectory)
65    {
66  
67      if (baseDirectory == null)
68        // exception here is required since File constructor below
69        // does not fail when called with null value for baseDirectory.
70        // We would not be acting fail fast without throwing an exception here.
71        throw new IllegalStateException("baseDirectory was null");
72      return new File(baseDirectory, "build");
73    }
